1) Adjust CMake file of piquasso boost for the new CPP and MPFR (per rp_DFE2)
    
2) Build piquassoboost
cd piquassoboost
python3 setup.py build_ext
cd ..

3) Build FPGA SIM and DFE:
cd workspace
make SIM
make DFE
cd ..

4) Run simulator (must run from workspace):
cd workspace
make CPU
#make CPUSIM - only run simulator without compiling CPU (if already compiled)
cd ..

5) Run DFE (must run from piquassoboost):
cd piquassoboost
env LD_LIBRARY_PATH=/home/$USER/workspace/PermanentGlynnCPU/dist/release/lib:$LD_LIBRARY_PATH python3 permanent_benchmark.py
cd ..
