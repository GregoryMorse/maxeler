piquassoboost/sampling/GlynnPermanentCalculator_Wrapper.hpp -> ~/piquassoboost/sampling/GlynnPermanentCalculator_Wrapper.hpp
piquassoboost/sampling/source/GlynnPermanentCalculatorDFE.cpp -> ~/piquassoboost/sampling/source/GlynnPermanentCalculatorDFE.cpp

piquassoboost/sampling/source/GlynnPermanentCalculatorRepeatedDFE.h -> ~/piquassoboost/sampling/source/GlynnPermanentCalculatorRepeatedDFE.h

1) Adjust CMake files of piquasso boost

Add to CMakeLists.txt of piquassoboost under "list(APPEND piquassoboost_files":
    ${PROJECT_SOURCE_DIR}/piquassoboost/sampling/source/GlynnPermanentCalculatorInf.cpp
    ${PROJECT_SOURCE_DIR}/piquassoboost/sampling/source/GlynnPermanentCalculatorDFE.cpp
    ${PROJECT_SOURCE_DIR}/piquassoboost/sampling/source/GlynnPermanentCalculatorRepeatedDFE.cpp

Add to CMakeLists.txt of piquassoboost/ under "target_link_libraries(piquassoboost PRIVATE":
    "/opt/anaconda3/lib/libgmp.so"
    "/opt/anaconda3/lib/libmpfr.so"
    
Add to CMakeLists.txt of piquassoboost/piquassoboost/sampling under "target_include_directories(Boson_Sampling_Utilities_wrapper PUBLIC":
                            /opt/anaconda3/include

2) Build piquassoboost
cd piquassoboost
cd ..

3) Build FPGA SIM and DFE:
cd workspace
make SIM
make DFE
cd ..

4) Run simulator (must run from workspace):
cd workspace
make CPU
#make CPUSIM - only run simulator without compiling CPU (if already compiled)
cd ..

5) Run DFE (must run from piquassoboost):
cd piquassoboost
env LD_LIBRARY_PATH=/home/$USER/workspace/PermanentGlynnCPU/dist/release/lib:$LD_LIBRARY_PATH python3 permanent_benchmark.py
cd ..
